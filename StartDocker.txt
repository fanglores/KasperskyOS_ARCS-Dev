#BUILD KOS IMAGE
sudo docker image build --file Dockerfile --tag kos .

#BASIC DOCKER RUN COMMAND
sudo docker run -it -v /home/user/Загрузки/KOS-folder/shared:/workspace --rm kos bash

#TCP DOCKER KOS RUN COMMAND
sudo docker run -it -v /home/user/Загрузки/KOS-folder/shared:/workspace --device /dev/net/tun:/dev/net/tun --privileged --rm kos bash

#TEST DOCKER RUN COMMAND
sudo docker run -it -v /home/user/Загрузки/KOS-folder/shared:/workspace -v /dev/mem:/dev/mem --rm kos bash
